示例1：Oozie调度Sqoop抽取数据到Hive表
================================================================================
## 1.目录结构
workflow的目录结构如下：
```
zjimee_fhd001/JTGK/apps/
+ job.properties
+ workflow.xml
```
workflow使用hive用户上传到HDFS上的/user/hive目录下。
```shell
hadoop fs -put zjimee_fuhd /user/hive/zjimee_fuhd
```

## 2.job.properties
```ini
nameNode=hdfs://server01.zjimee.com:8020
jobTracker=server01.zjimee.com:8050
queueName=default
appRoot=zjimee_fhd001

jdbcUrl=jdbc:mysql://xxx.xxx.xxx.xxx:3306/road_monitor
username=root
password=123

oozie.use.system.libpath=true
oozie.libpath=${nameNode}/user/oozie/share/lib/lib_20190709165044/hive2

oozie.wf.application.path=${nameNode}/user/${user.name}/${appRoot}/JTGK/apps
```

## 3.workflow.xml
```
<?xml version="1.0" encoding="UTF-8"?>

<workflow-app xmlns="uri:oozie:workflow:0.5" name="jtgk-wf">

    <start to="sqoop_capital_construction_job"/>
    <action name="sqoop_capital_construction_job">
        <sqoop xmlns="uri:oozie:sqoop-action:0.2">
            <job-tracker>${jobTracker}</job-tracker>
            <name-node>${nameNode}</name-node>
            <prepare>
                <delete path="${nameNode}/user/${wf:user()}/capital_construction"/>
            </prepare>
            <configuration>
                <property>
                    <name>mapred.job.queue.name</name>
                    <value>${queueName}</value>
                </property>
            </configuration>
            <arg>import</arg>
            <arg>--connect</arg>
            <arg>${jdbcUrl}</arg>
            <arg>--username</arg>
            <arg>${username}</arg>
            <arg>--password</arg>
            <arg>${password}</arg>
            <arg>--table</arg>
            <arg>capital_construction</arg>
            <arg>--hive-import</arg>
            <arg>--hive-overwrite</arg>
            <arg>--hive-database</arg>
            <arg>road_monitor</arg>
            <arg>--hive-table</arg>
            <arg>ods_road_monitor_capital_construction_1</arg>
            <arg>--as-orcfile</arg>
            <arg>-m</arg>
            <arg>1</arg>
        </sqoop>
        <ok to="end"/>
        <error to="fail01"/>
    </action>
    <kill name="fail01">
        <message>jtgk-fork01-node to jtgk-join01-node,Sqoop failed, error message[${wf:errorMessage(wf:lastErrorNode())}]</message>
    </kill>
    <end name="end"/>
</workflow-app>
```

## 4.Oozie命令
```sql
--hive用户下执行命令
oozie job -oozie http://server01.zjimee.com:11000/oozie -config zjimee_fuhd/JTGK/apps/job.properties -run
```
