Oozie调度Sqoop动作一定要使用sqoop-freeform方式
===================================================================================
## 错误示例
```xml
<?xml version="1.0" encoding="UTF-8"?>

<workflow-app xmlns="uri:oozie:workflow:0.5" name="sqoop-wf-002">
    <start to="sqoop-node"/>

    <action name="sqoop-node">
        <sqoop xmlns="uri:oozie:sqoop-action:0.3">
            <job-tracker>${jobTracker}</job-tracker>
            <name-node>${nameNode}</name-node>
            <prepare>
                <delete path="${nameNode}/warehouse/tablespace/external/hive/road_monitor.db/ods_road_monitor_divide_info_1/"/>
            </prepare>
            <configuration>
                <property>
                    <name>mapred.job.queue.name</name>
                    <value>${queueName}</value>
                </property>
            </configuration>
            <command>sqoop import --connect jdbc:mysql://119.23.213.192:3306/road_monitor?serverTimezone=UTC --table divide_info --username root --password 123 --target-dir /warehouse/tablespace/external/hive/road_monitor.db/ods_road_monitor_divide_info_1 -m 1</command>
        </sqoop>
        <ok to="end"/>
        <error to="fail"/>
    </action>

    <kill name="fail">
        <message>Sqoop failed, error message[${wf:errorMessage(wf:lastErrorNode())}]</message>
    </kill>
    <end name="end"/>
</workflow-app>
```
如果把sqoop命令通过<command>写在一行，会出错！！！原因没有细查。

## 正确示例
通过`sqoop-freeform`方式是正确的，验证过了！！
```xml
<?xml version="1.0" encoding="UTF-8"?>

<workflow-app xmlns="uri:oozie:workflow:0.5" name="sqoop-wf-004">
    <start to="sqoop-node"/>

    <action name="sqoop-node">
        <sqoop xmlns="uri:oozie:sqoop-action:0.3">
            <job-tracker>${jobTracker}</job-tracker>
            <name-node>${nameNode}</name-node>
            <prepare>
                <delete path="${nameNode}/warehouse/tablespace/external/hive/road_monitor.db/ods_road_monitor_divide_info_1"/>
            </prepare>
            <configuration>
                <property>
                    <name>mapred.job.queue.name</name>
                    <value>${queueName}</value>
                </property>
            </configuration>
            <arg>import</arg>
            <arg>--connect</arg>
            <arg>jdbc:mysql://119.23.213.192:3306/road_monitor?serverTimezone=UTC</arg>
            <arg>--table</arg>
            <arg>divide_info</arg>
            <arg>--username</arg>
            <arg>root</arg>
            <arg>--password</arg>
            <arg>123</arg>
            <arg>--target-dir</arg>
            <arg>/warehouse/tablespace/external/hive/road_monitor.db/ods_road_monitor_divide_info_1</arg>
            <arg>-m</arg>
            <arg>1</arg>
        </sqoop>
        <ok to="end"/>
        <error to="fail"/>
    </action>

    <kill name="fail">
        <message>Sqoop failed, error message[${wf:errorMessage(wf:lastErrorNode())}]</message>
    </kill>
    <end name="end"/>
</workflow-app>
```
